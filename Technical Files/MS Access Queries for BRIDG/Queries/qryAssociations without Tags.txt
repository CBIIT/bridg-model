SELECT t_package.Name AS [Source Package], t_connectortypes.Connector_Type AS [Connector Type], t_object.Name AS [Start Class], t_connector.Name AS [Connector Name], t_object_1.Name AS [End Class], t_connectortag.PropertyID AS Tag
FROM t_package INNER JOIN (((t_object INNER JOIN (t_connectortypes INNER JOIN t_connector ON t_connectortypes.Connector_Type = t_connector.Connector_Type) ON t_object.Object_ID = t_connector.Start_Object_ID) INNER JOIN t_object AS t_object_1 ON t_connector.End_Object_ID = t_object_1.Object_ID) LEFT JOIN t_connectortag ON t_connector.Connector_ID = t_connectortag.ElementID) ON t_package.Package_ID = t_object.Package_ID
WHERE (((t_package.Name)="Common Sub-Domain" OR (t_package.Name)="Protocol Representation Sub-Domain" OR (t_package.Name)="Adverse Event Sub-Domain" OR (t_package.Name)="Regulatory Sub-Domain" OR (t_package.Name)="Study Conduct Sub-Domain") AND ((t_connectortag.PropertyID) Is Null) AND ((t_object.Object_Type)="Class") AND ((t_object_1.Object_Type)="Class"))
ORDER BY t_package.Name, t_connectortypes.Connector_Type, t_object.Name;
