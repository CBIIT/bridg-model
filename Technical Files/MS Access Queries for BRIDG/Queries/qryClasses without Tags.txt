SELECT t_package.Name AS Package, t_object.Name AS Class, t_objectproperties.PropertyID AS Tag
FROM t_package INNER JOIN (t_object LEFT JOIN t_objectproperties ON t_object.Object_ID = t_objectproperties.Object_ID) ON t_package.Package_ID = t_object.Package_ID
WHERE  (((t_package.Name)="Protocol Representation Sub-domain" Or (t_package.Name)="Common Sub-domain" Or (t_package.Name)="Adverse Event Sub-domain" Or (t_package.Name)="Regulatory Sub-domain" Or (t_package.Name)="Study Conduct Sub-domain") AND ((t_objectproperties.PropertyID) Is Null) AND ((t_object.Object_Type)="Class"))
ORDER BY t_object.Name;
