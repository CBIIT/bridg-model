SELECT t_object.Name AS [Class Name], t_attribute.Name AS [Attribute Name], t_attribute.type AS Type, Mid(t_attribute.notes,1,250) AS [Class Definition Part 1]
FROM t_package INNER JOIN (t_object INNER JOIN t_attribute ON t_object.object_id = t_attribute.object_id) ON t_package.Package_ID = t_object.Package_ID
WHERE (((t_package.Name)="Common Sub-domain" Or (t_package.Name)="Adverse Event Sub-domain" Or (t_package.Name)="Regulatory Sub-domain" Or (t_package.Name)="Protocol Representation Sub-domain" Or (t_package.Name)="Study Conduct Sub-domain") AND ((t_object.Object_Type)="Class") AND ((UCase([t_object].[name])) Not Like "DELETED*"))
ORDER BY t_attribute.Name, t_object.Name;
