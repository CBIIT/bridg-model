SELECT t_package.Name AS Package, t_object.Name AS Class, t_attribute.Name AS Attribute, t_attribute.Notes AS [Attribute Defintion], t_attribute.Type AS [Data Type], t_attribute.Derived, t_attribute.LowerBound, t_attribute.UpperBound, t_attribute.Pos AS [Attribute Position]
FROM t_package INNER JOIN (t_object INNER JOIN t_attribute ON t_object.Object_ID = t_attribute.Object_ID) ON t_package.Package_ID = t_object.Package_ID
WHERE (((t_package.Name)="Protocol Representation Sub-domain" Or (t_package.Name)="Common Sub-domain" Or (t_package.Name)="Adverse Event Sub-domain" Or (t_package.Name)="Regulatory Sub-domain" Or (t_package.Name)="Study Conduct Sub-domain") AND ((t_object.Object_Type)="Class"))
ORDER BY t_package.Name, t_object.Name, t_attribute.Pos;
